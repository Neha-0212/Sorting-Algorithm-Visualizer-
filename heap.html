heap Sort

Heap Sort is a sorting algorithm that organizes elements in an array into a binary heap, a tree-like data structure where the value of each parent node is greater than or equal to the values of its children (for max-heap). It then repeatedly extracts the maximum element from the heap and places it at the end of the array, gradually building up the sorted list. This process continues until all elements have been extracted and the array is fully sorted.

Heap Sort is efficient with a time complexity of O(n log n) and is particularly useful for sorting large datasets due to its ability to sort in-place, meaning it requires only a constant amount of extra memory beyond the input array.

Algorithm

Step I : Build a max-heap (or min-heap) from the input list.
Step II : Swap the root element of the heap with the last element in the unsorted region.
Step III : Decrease the size of the unsorted region by one.
Step IV : Restore the max-heap (or min-heap) property.
Step V : Repeat steps 2-4 until the entire list is sorted.

When To Use

When prioritizing efficiency, especially for large datasets.
When stability in sorting order is not critical.
When additional memory usage is not a major concern.

Time & Space Complex

Case Complexity
Best Case (Ω) O(nlogn)
Average Case (Θ) O(nlogn)
Worst Case (O) O(nlogn)
Space O(1)

                def heapify(arr, n, i):
                    largest = i  # Initialize largest as root
                    left = 2 * i + 1  # Left child
                    right = 2 * i + 2  # Right child

                    # If left child is larger than root
                    if left < n and arr[left] > arr[largest]:
                        largest = left

                    # If right child is larger than largest so far
                    if right < n and arr[right] > arr[largest]:
                        largest = right

                    # If largest is not root
                    if largest != i:
                        # Swap the largest element with the root
                        arr[i], arr[largest] = arr[largest], arr[i]

                        # Recursively heapify the affected sub-tree
                        heapify(arr, n, largest)


                def heapSort(arr):
                    n = len(arr)

                    # Build a maxheap.
                    # Since last parent will be at ((n//2)-1) we can start at that location.
                    for i in range(n // 2 - 1, -1, -1):
                        heapify(arr, n, i)

                    # One by one extract elements
                    for i in range(n - 1, 0, -1):
                        arr[i], arr[0] = arr[0], arr[i]  # swap
                        heapify(arr, i, 0)
            
                    // Function to heapify a subtree rooted at node i
                    void heapify(int arr[], int n, int i) {
                        int largest = i; // Initialize largest as root
                        int left = 2 * i + 1; // Left child
                        int right = 2 * i + 2; // Right child
                    
                        // If left child is larger than root
                        if (left < n && arr[left] > arr[largest])
                            largest = left;
                    
                        // If right child is larger than largest so far
                        if (right < n && arr[right] > arr[largest])
                            largest = right;
                    
                        // If largest is not root
                        if (largest != i) {
                            // Swap the largest element with the root
                            int temp = arr[i];
                            arr[i] = arr[largest];
                            arr[largest] = temp;
                    
                            // Recursively heapify the affected sub-tree
                            heapify(arr, n, largest);
                        }
                    }
                    
                    // Function to perform heap sort
                    void heapSort(int arr[], int n) {
                        // Build heap (rearrange array)
                        for (int i = n / 2 - 1; i >= 0; i--)
                            heapify(arr, n, i);
                    
                        // One by one extract an element from heap
                        for (int i = n - 1; i > 0; i--) {
                            // Move current root to end
                            int temp = arr[0];
                            arr[0] = arr[i];
                            arr[i] = temp;
                    
                            // Call max heapify on the reduced heap
                            heapify(arr, i, 0);
                        }
                    }

                function heapSort(arr) {
                    const n = arr.length;
                
                    // Build heap (rearrange array)
                    for (let i = Math.floor(n / 2) - 1; i >= 0; i--)
                        heapify(arr, n, i);
                
                    // One by one extract an element from heap
                    for (let i = n - 1; i > 0; i--) {
                        // Move current root to end
                        [arr[0], arr[i]] = [arr[i], arr[0]];
                
                        // Call max heapify on the reduced heap
                        heapify(arr, i, 0);
                    }
                }               
            
                public class HeapSort {
                    public static void heapSort(int[] arr) {
                        int n = arr.length;

                        // Build heap (rearrange array)
                        for (int i = n / 2 - 1; i >= 0; i--)
                            heapify(arr, n, i);

                        // One by one extract an element from heap
                        for (int i = n - 1; i > 0; i--) {
                            // Move current root to end
                            int temp = arr[0];
                            arr[0] = arr[i];
                            arr[i] = temp;

                            // Call max heapify on the reduced heap
                            heapify(arr, i, 0);
                        }
                    }

                    // To heapify a subtree rooted with node i which is an index in arr[]
                    static void heapify(int[] arr, int n, int i) {
                        int largest = i; // Initialize largest as root
                        int left = 2 * i + 1; // Left child
                        int right = 2 * i + 2; // Right child

                        // If left child is larger than root
                        if (left < n && arr[left] > arr[largest])
                            largest = left;

                        // If right child is larger than largest so far
                        if (right < n && arr[right] > arr[largest])
                            largest = right;

                        // If largest is not root
                        if (largest != i) {
                            // Swap the largest element with the root
                            int temp = arr[i];
                            arr[i] = arr[largest];
                            arr[largest] = temp;

                            // Recursively heapify the affected sub-tree
                            heapify(arr, n, largest);
                        }
                    }
                }
